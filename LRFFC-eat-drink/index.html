function makeTrial() { const targetType = state.mode === "eat" ? "food" : "drink"; const nonTargetTypes = state.mode === "eat" ? ["drink", "other"] : ["food", "other"]; // Pick ONE correct target const target = pickRandom( ITEM_BANK.filter(i => i.type === targetType), 1 )[0]; // Pick distractors ONLY from non-target types const distractors = pickRandom( ITEM_BANK.filter(i => nonTargetTypes.includes(i.type)), CONFIG.fieldSize - 1 ); return { id: `${state.mode}-${target.id}-${Math.random()}`, targetId: target.id, choices: shuffle([target, ...distractors]), firstAttemptWrong: false }; }
